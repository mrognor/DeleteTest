478  if (ltm->tm_sec < 10) currentTime += "0";
479  currentTime += std::to_string(ltm->tm_sec);
480 
481  auto currentTimeSec = std::chrono::duration_cast<std::chrono::seconds>(now.time_since_epoch()).count();
482  auto currentTimeT = std::chrono::duration_cast<T>(now.time_since_epoch()).count();
483  std::string stringTimeT = std::to_string(currentTimeT);
484 
485  int len = currentTimeT / currentTimeSec;
486  switch (len)
487  {
488  // Milliseconds
489  case 1000:
490  currentTime = currentTime + "." + stringTimeT.substr(stringTimeT.length() - 3);
491  break;
492  // Microseconds
493  case 1000000:
494  currentTime = currentTime + "." + stringTimeT.substr(stringTimeT.length() - 6);
495  break;
496  // Nanoseconds
497  case 1000000000:
